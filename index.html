package framework;
import framework.utils.WaitingUtils;
import framework.utils.Logger;
public abstract class BaseForm {
private BaseElement uniqElement;
private String formName;
protected BaseForm(BaseElement uniqElement, String formName) {
this.uniqElement = uniqElement;
this.formName = formName;
}
public boolean isFormOpen() {
if(uniqElement.isPresentUniqElement())
Logger.logInfo("open " + formName + ".");
else
Logger.logWarn("not open " + formName + ".");
return uniqElement.isPresentUniqElement();
}
public boolean isFormClose() {
if(WaitingUtils.invisibilityOfElementLocated(uniqElement.getLocator()))
Logger.logInfo("close " + formName + ".");
else
Logger.logWarn("not close " + formName + ".");
return WaitingUtils.invisibilityOfElementLocated(uniqElement.getLocator());
}
}
